## SearchTopResponse

```ts
type SearchTopResponse = {
  data: {
    top?: (
      | Song
      | BaseSong
      | Album
      | BaseAlbum
      | Artist
      | BaseArtist
      | Playlist
      | BasePlaylist
      | Station
      | BaseStation
    )[];
    songs?: (Song | BaseSong)[];
    albums?: (Album | BaseAlbum)[];
    artists?: (Artist | BaseArtist)[];
    playlists?: (Playlist | BasePlaylist)[];
    stations?: (Station | BaseStation)[];
  };
};
```

```ts

    code: number;
    title: string;
    message: string;
    trace_id: string | null;
  };
};
```
